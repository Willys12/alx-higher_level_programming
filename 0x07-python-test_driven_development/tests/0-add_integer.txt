#!/usr/bin/python3
"""Defines an integer addition function.

This module provides a function, add_integer, which returns the integer sum of
two numeric values. If either of the arguments is a float, it is typecasted to
an integer before the addition is performed.

Raises:
    TypeError: If either of the arguments is a non-integer and non-float.
"""

def add_integer(a, b=98):
    """Return the integer sum of two numeric values.

    Args:
        a (int or float): The first numeric value.
        b (int or float, optional): The second numeric value. Defaults to 98.

    Returns:
        int: The integer sum of a and b.

    Raises:
        TypeError: If either of the arguments is a non-integer and non-float.

    Example:
        >>> add_integer(5, 3.5)
        8
    """
    if not isinstance(a, (int, float)):
        raise TypeError("a must be an integer or float")
    if not isinstance(b, (int, float)):
        raise TypeError("b must be an integer or float")

    return int(a) + int(b)
